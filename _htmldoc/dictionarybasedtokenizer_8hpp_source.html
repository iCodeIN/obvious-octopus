35  {
36  auto len = i - tokenBoundaries[tokenBoundaries.size() - 1];
37  auto token = s.substr(tokenBoundaries[tokenBoundaries.size()-1], len);
38  if(this->isSet(IGNORE_CASE))
39  {
40  token = util::string::toUpper(token);
41  }
42 
43  // if the token is a valid prefix, continue building the token
44  if(isPrefix(token))
45  {
46  // std::cout << token << ", pos : " << i << ", valid prefix, continue" << std::endl;
47  continue;
48  }
49  // if the token is a word, mark the boundary
50  else if(isWord(token))
51  {
52  // std::cout << token << ", pos : " << i << ", valid word, push boundary" << std::endl;
53  tokenBoundaries.push_back(i);
54  continue;
55  }
56  // unhappy flow
57  else
58  {
59  /* The code that follows relies on there being a prefix of the current token
60  if that is not the case, we deal with it here.
61  */
62  if(token.size() == 1)
63  {
64  // std::cout << token << ", pos : " << i << ", single char token, push boundary" << std::endl;
65  tokenBoundaries.push_back(i);
66  }
67  else
68  {
69  /* If the prefix of the current token is valid prefix (or better, a word)
70  we mark the boundary, and go back one character.
71  */
72  auto prevToken = token.substr(0, token.size() - 1);
73  if(isPrefix(prevToken) || isWord(prevToken))
74  {
75  // std::cout << token << ", pos : " << i << ", prev token was valid prefix, or valid word, push boundary, resume from " << (i-1) << std::endl;
76  tokenBoundaries.push_back(i - 1);
77  i--;
78  }
79  else
80  {
81  // std::cout << token << ", pos : " << i << ", no useful information, push boundary" << std::endl;
82  tokenBoundaries.push_back(i);
83  }
84  }
85  }
86  }
