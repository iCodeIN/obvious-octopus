138  int longestPath(const IGraph<T> &graph, const T &source) const
139  {
140  // graph should contain the source node
141  assert(graph.hasVertex(source));
142 
143  // nodes that have already been explored
144  std::set<T> visited;
145 
146  // candidate nodes for exploration
147  using PathInfoType = std::tuple<const T, int>;
148  std::stack<PathInfoType> toVisit;
149 
150  // put query node up for exploration
151  toVisit.push(std::make_tuple(source, 0));
152 
153  // main loop
154  auto maxLen = 0;
155  while(!toVisit.empty())
156  {
157  auto curr = toVisit.top();
158  toVisit.pop();
159 
160  auto id = std::get<0>(curr);
161  auto len = std::get<1>(curr);
162  maxLen = std::max(maxLen, len);
163 
164  if(std::find(visited.begin(), visited.end(), id) != visited.end())
165  {
166  continue;
167  }
168 
169  // graph is not supposed to contain cycles
170  assert(!(maxLen > 0 && id == source));
171 
172  visited.insert(id);
173 
174  for(auto &target : graph.outgoing(id))
175  {
176  toVisit.push(std::make_tuple(target, len + 1));
177  }
178  }
179 
180  return maxLen;
181  }
