256  static void fromTree(XML::Element &element, const boost::property_tree::ptree &tree)
257  {
258  /* The first level of a boost::property_tree::ptree is always empty.
259  When the recursion 'fromTree' start, the element does not have a name.
260  This condition enables the algorithm to skip the empty root and move on to the 'real' root of the XML tree.
261  */
262  if(element.getName().empty())
263  {
264  auto realRoot = *(tree.begin());
265  element.setName(realRoot.first);
266  fromTree(element, realRoot.second);
267  return;
268  }
269 
270  // process children
271  for(auto child : tree)
272  {
273  auto name = child.first;
274 
275  // if the name is '<xmlattr>' then we are dealing with XML attributes
276  if(name == "<xmlattr>")
277  {
278  // process attributes
279  for(auto attrChild : child.second)
280  {
281  element.setAttribute(attrChild.first, attrChild.second.get_value<std::string>());
282  }
283  }
284  // else we are dealing with regular XML child-nodes
285  else
286  {
287  std::unique_ptr<XML::Element> childPtr = std::unique_ptr<XML::Element>(new XML::Element(name));
288  fromTree(*childPtr.get(), child.second);
289  element.add(std::move(childPtr));
290  }
291  }
