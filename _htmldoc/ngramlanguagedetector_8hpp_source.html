124  {

135  // cosine similarity metric with naive discounted smoothing
136  auto alpha = 0.001;
137  auto terms = 0.0;
138  auto squaresA = 0.0;
139  auto squaresB = 0.0;
140  for(auto &ngram : ngramsUnion)
141  {
142  auto f0 = lm0.find(ngram) != lm0.end() ? lm0.find(ngram)->second * (1.0 - alpha) : alpha;
143  auto f1 = lm1.find(ngram) != lm1.end() ? lm1.find(ngram)->second * (1.0 - alpha) : alpha;
144  terms += f0 * f1;

147  }
148  return terms / (sqrt(squaresA) * sqrt(squaresB));
