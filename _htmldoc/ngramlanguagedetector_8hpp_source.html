128  // cosine similarity metric with naive discounted smoothing
129  auto alpha = 0.001;
130  auto terms = 0.0;
131  auto squaresA = 0.0;
132  auto squaresB = 0.0;
133  for(auto &ngram : ngramsUnion)
134  {
135  auto f0 = lm0.find(ngram) != lm0.end() ? lm0.find(ngram)->second * (1.0 - alpha) : alpha;
136  auto f1 = lm1.find(ngram) != lm1.end() ? lm1.find(ngram)->second * (1.0 - alpha) : alpha;
137  terms += f0 * f1;

140  }
141  return terms / (sqrt(squaresA) * sqrt(squaresB));
