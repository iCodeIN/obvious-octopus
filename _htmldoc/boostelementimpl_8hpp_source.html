193  static void toTree(const XML::IElement* e, const std::string &path, boost::property_tree::ptree &tree)
194  {
195  std::string newPath = path.empty() ? e->getName() : path + "." + e->getName();
196 
197  // put text
198  auto &it = tree.add(newPath, e->getText());
199 
200  // process attributes
201  for(auto pair : e->getAttributes())
202  {
203  it.add("<xmlattr>." + pair.first, pair.second);
204  }
205 
206  // process children
207  int i = 0;
208  while(e->hasChild(i))
209  {
210  toTree(&(e->getChild(i)), newPath, tree);
211  i++;
212  }
213  }

217  static void fromTree(XML::IElement* e, const boost::property_tree::ptree &tree)
218  {
219  /* The first level of a boost::property_tree::ptree is always empty.
220  When the recursion 'fromTree' start, the element does not have a name.
221  This condition enables the algorithm to skip the empty root and move on to the 'real' root of the XML tree.
222  */
223  if(e->getName().empty())
224  {
225  auto realRoot = *(tree.begin());
226  e->setName(realRoot.first);
227  fromTree(e, realRoot.second);
228  return;
229  }
230 
231  // process children
232  for(auto child : tree)
233  {
234  auto name = child.first;
235 
236  // if the name is '<xmlattr>' then we are dealing with XML attributes
237  if(name == "<xmlattr>")
238  {
239  // process attributes
240  for(auto attrChild : child.second)
241  {
242  e->setAttribute(attrChild.first, attrChild.second.get_value<std::string>());
243  }
244  }
245  // else we are dealing with regular XML child-nodes
246  else
247  {
248  auto childPtr = std::unique_ptr<XML::IElement>(new XML::BoostElementImpl(name));
249  fromTree(childPtr.get(), child.second);
250  e->add(std::move(childPtr));
251  }
252  }
